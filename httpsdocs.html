<div id="chart" style="width: 800px; height: 400px;"></div>


<script>

google.charts.load('current', { packages: ['corechart'] }); google.charts.setOnLoadCallback(drawChart);


function drawChart() {

const query = new google.visualization.Query( 'https://docs.google.com/spreadsheets/d/1n9MA55yNl9jNG-

pbX4N5DztGf83afArwTdung64sxV4/gviz/tq?sheet=Running_totals'

);


query.send(response => {

const data = response.getDataTable();

const chart = new google.visualization.LineChart( document.getElementById('chart')

);

chart.draw(data, { title: 'Sales Over Time' });

});

}

</script>
