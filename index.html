<script>

google.charts.load('current', { packages: ['corechart'] }); google.charts.setOnLoadCallback(drawCharts);


function drawCharts() {

const sheetId = '1n9MA55yNl9jNG-pbX4N5DztGf83afArwTdung64sxV4'; const sheetName = 'Running Totals';


const query = new google.visualization.Query(

`https://docs.google.com/spreadsheets/d/${sheetId}/gviz/tq?sheet=${encodeURICompo nent(sheetName)}`

);


query.send(response => { if (response.isError()) {

console.error(response.getMessage()); return;

}


const rawData = response.getDataTable(); drawTotalChart(rawData); drawTypeChart(rawData);

});

}


function drawTotalChart(data) {

const totalData = new google.visualization.DataTable(); totalData.addColumn('string', 'Date'); totalData.addColumn('number', 'Hot Dogs');


const totals = {};


for (let i = 0; i < data.getNumberOfRows(); i++) { const date = data.getValue(i, 0);

const qty = data.getValue(i, 1); totals[date] = (totals[date] || 0) + qty;

}


for (const date in totals) { totalData.addRow([date, totals[date]]);

}


const chart = new google.visualization.ColumnChart( document.getElementById('total_chart')

);

chart.draw(totalData, {

title: 'Total Hot Dogs Consumed by Date', legend: 'none',

hAxis: { title: 'Date' }, vAxis: { title: 'Hot Dogs' }

});

}


function drawTypeChart(data) { const typeTotals = {};


for (let i = 0; i < data.getNumberOfRows(); i++) { const type = data.getValue(i, 2);

const qty = data.getValue(i, 1); typeTotals[type] = (typeTotals[type] || 0) + qty;

}


const typeData = new google.visualization.DataTable(); typeData.addColumn('string', 'Type'); typeData.addColumn('number', 'Quantity');


for (const type in typeTotals) { typeData.addRow([type, typeTotals[type]]);

}


const chart = new google.visualization.PieChart( document.getElementById('type_chart')

);


chart.draw(typeData, { title: 'Hot Dogs by Type'

});

}
